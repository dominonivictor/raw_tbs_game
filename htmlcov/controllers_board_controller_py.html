208def calculate_dijkstras(start_tile, board, game_grid, step_cost_function, n_moves): 

209 #calculate_cost_to_move -> for moving, calculate_cost_to_attack -> for attacking 

210 x, y = start_tile.grid_x, start_tile.grid_y 

211 closed_list = [] 

212 open_dict = {(x, y): {"dist": 0, "via": None}} 

213 tile_xy = True #just for getting in the loop 

214 while(tile_xy): 

215 tile_xy = min(open_dict, key=lambda k: open_dict[k]["dist"]) if open_dict else None 

216 if tile_xy: 

217 tile_dict = open_dict.pop(tile_xy) 

218 tile_neighbors = get_tile_neighbors(tile_xy=tile_xy, max_size=board.grid_size, 

219 game_grid=game_grid, closed_list=closed_list) 

220 

221 for neigh_xy in tile_neighbors: 

222 neigh_dict = open_dict.get(neigh_xy, None) 

223 cost_for_step = step_cost_function(pos=neigh_xy, game_grid=game_grid) 

224 dist_now = tile_dict["dist"] 

225 final_dist = dist_now + cost_for_step 

226 if final_dist <= n_moves: 

227 if(not neigh_dict or final_dist < neigh_dict["dist"]): 

228 open_dict[neigh_xy] = {"dist": final_dist, "via": tile_xy} 

229 

230 closed_list.append(tile_xy) 

231 

232 return closed_list

241def get_tile_neighbors(tile_xy, max_size, game_grid, closed_list): 

242 neighbors = [] 

243 i, j = tile_xy 

244 if i - 1 >= 0: 

245 neighbors.append((i-1, j)) 

246 if i + 1 <= max_size - 1: 

247 neighbors.append((i+1, j)) 

248 if j - 1 >= 0: 

249 neighbors.append((i, j-1)) 

250 if j + 1 <= max_size - 1: 

251 neighbors.append((i, j+1)) 

252 

253 for coords in neighbors: 

254 x, y = coords 

255 if(game_grid[x][y].is_blocked): 

256 neighbors.remove((x, y)) 

257 

258 return neighbors
